```
def test_prompt(template_name, template, **kwargs):
    """Тестирует один промпт"""
    print(f"\n{'='*50}")
    print(f"ТЕСТИРУЕМ: {template_name}")
    print(f"{'='*50}")

        {'role': 'user', 'content': prompt}
    ])

    answer = response['message']['content']

    return answer
```

```
import ollama
import time

def test_speed(model, options=None):
    """Замеряет скорость с заданными опциями"""
    if options is None:
        options = {}

    prompt = "Почему машина не заводится? Ответь кратко."

    start = time.time()
    response = ollama.chat(
        model=model,
        messages=[{'role': 'user', 'content': prompt}],
        options=options  # тут передаются параметры оптимизации
    )
    latency = time.time() - start

    tokens = len(response['message']['content'].split())
    return latency, tokens

print(" ТЕСТИРОВАНИЕ ОПТИМИЗАЦИЙ")
print("="*50)

# Тест 1: Базовая настройка (без оптимизации)
lat1, tok1 = test_speed('llama3.1:8b')
print(f"\n Базовая настройка:")
print(f"  Время: {lat1:.2f} сек")
print(f"  Скорость: {tok1/lat1:.1f} токенов/сек")

# Тест 2: С оптимизациями
options = {
    'num_ctx': 1024,        
    'num_batch': 256,       
    'num_thread': 8,         
    'f16_kv': True,          
    'use_mmap': True,        
    'low_vram': True         
}

lat2, tok2 = test_speed('llama3.1:8b', options)
print(f"\n С оптимизацией:")
print(f"  Время: {lat2:.2f} сек")
print(f"  Скорость: {tok2/lat2:.1f} токенов/сек")

# Результат
improvement = ((lat1 - lat2) / lat1) * 100
print(f"\n Улучшение: {improvement:.1f}% быстрее")
```

```
Диагностика двигателя:
1. Если машина не заводится, проверьте аккумулятор (напряжение должно быть 12.5-12.7В)
2. Если стартер крутит, но двигатель не схватывает — проверьте топливную систему
3. Если слышны щелчки при повороте ключа — проблема в стартере или аккумуляторе

Замена масла:
1. Для бензиновых двигателей — каждые 8-10 тыс. км
2. Для дизельных — каждые 7-8 тыс. км
3. Используйте масло 5W30 или 5W40 в зависимости от сезона

Перегрев двигателя:
1. Проверьте уровень антифриза
2. Убедитесь, что работает вентилятор охлаждения
3. Возможна неисправность термостата

Тормозная система:
1. Тормозные колодки служат 20-30 тыс. км
2. При скрипе тормозов проверьте износ колодок
3. Тормозную жидкость менять каждые 2 года
```

```
import os
import warnings
warnings.filterwarnings('ignore')  

from langchain_chroma import Chroma
from langchain_community.embeddings import HuggingFaceEmbeddings
from langchain_community.llms import Ollama
from langchain_text_splitters import RecursiveCharacterTextSplitter
from langchain_community.document_loaders import TextLoader
from datetime import datetime

class CarRAG:
    def __init__(self, model_name="llama3.1:8b", persist_dir="./car_db"):
        self.model_name = model_name
        self.persist_dir = persist_dir

        # Эмбеддинги на русском
        self.embeddings = HuggingFaceEmbeddings(
            model_name="intfloat/multilingual-e5-small",
            model_kwargs={'device': 'cpu'}
        )

        self.vectorstore = None
        self.llm = Ollama(model=self.model_name)

    def create_db(self):
        """Создаёт базу знаний из файла"""
        print(" Создаём базу знаний...")

        # Загружаем документ
        loader = TextLoader("car_knowledge.txt", encoding='utf-8')
        docs = loader.load()

        # Разбиваем на кусочки
        splitter = RecursiveCharacterTextSplitter(
            chunk_size=300,
            chunk_overlap=50
        )
        chunks = splitter.split_documents(docs)

        # Создаём векторную базу (persist больше не нужен)
        self.vectorstore = Chroma.from_documents(
            documents=chunks,
            embedding=self.embeddings,
            persist_directory=self.persist_dir
        )
        print(f" Создано {len(chunks)} фрагментов")

    def search(self, query, k=2):
        """Ищет похожие документы"""
        if not self.vectorstore:
            if os.path.exists(self.persist_dir):
                self.vectorstore = Chroma(
                    persist_directory=self.persist_dir,
                    embedding_function=self.embeddings
                )
            else:
                self.create_db()

        docs = self.vectorstore.similarity_search(query, k=k)
        return docs

    def ask(self, question):
        """Задаёт вопрос с поиском по документам"""
        # Ищем релевантные документы
        docs = self.search(question)

        # Формируем контекст из документов
        context = "\n\n".join([doc.page_content for doc in docs])

        # Формируем промпт с контекстом
        prompt = f"""Используй информацию из документов ниже, чтобы ответить на вопрос.
Если ответа нет в документах, скажи "Я не знаю".

Документы:
{context}

Вопрос: {question}

Ответ:"""

        # Отправляем в LLM
        start = datetime.now()
        response = self.llm.invoke(prompt)
        latency = (datetime.now() - start).total_seconds()

        return {
            'question': question,
            'answer': response,
            'latency': latency,
            'sources': [doc.metadata.get('source', 'unknown') for doc in docs]
        }

    def compare(self, question):
        """Сравнивает ответ с RAG и без"""
        print(f"\n Вопрос: {question}")
        print("-"*60)

        # Без RAG (просто LLM)
        start = datetime.now()
        simple_answer = self.llm.invoke(question)
        simple_time = (datetime.now() - start).total_seconds()

        print(f" БЕЗ RAG ({simple_time:.2f} сек):")
        print(simple_answer[:300] + "..." if len(simple_answer) > 300 else simple_answer)

        # С RAG
        rag_result = self.ask(question)
        print(f"\n С RAG ({rag_result['latency']:.2f} сек):")
        print(rag_result['answer'][:300] + "..." if len(rag_result['answer']) > 300 else rag_result['answer'])
        print(f"\n Источники: {', '.join(rag_result['sources'])}")
```

```
import os
import warnings
warnings.filterwarnings('ignore')
from datetime import datetime

# Импорты для RAG
from langchain_chroma import Chroma
from langchain_community.embeddings import HuggingFaceEmbeddings
from langchain_community.llms import Ollama
from langchain_text_splitters import RecursiveCharacterTextSplitter
from langchain_community.document_loaders import TextLoader

class SafeGuardRAG:
    """
    RAG-система с интегрированной моделью безопасности Granite Guardian.
    """
    def __init__(self, model_name="llama3.1:8b", guard_name="granite3-guardian:2b", persist_dir="./car_db_guard"):
        self.model_name = model_name
        self.guard_name = guard_name
        self.persist_dir = persist_dir

        # Инициализация эмбеддингов (для RAG)
        print(" Загрузка эмбеддингов...")
        self.embeddings = HuggingFaceEmbeddings(
            model_name="intfloat/multilingual-e5-small",
            model_kwargs={'device': 'cpu'}
        )
        self.vectorstore = None

        # Инициализация основной LLM (консьерж)
        print(f" Загрузка основной модели ({self.model_name})...")
        self.llm = Ollama(model=self.model_name)

        # Инициализация модели-охранника (bodyguard)
        print(f" Загрузка охранника ({self.guard_name})...")
        self.guard = Ollama(model=self.guard_name)
        print(" Система готова к работе!")

    def _get_vectorstore(self):
        """Создает или загружает векторную базу данных."""
        if self.vectorstore:
            return self.vectorstore

        if os.path.exists(self.persist_dir):
            print(" Загрузка существующей базы знаний...")
            self.vectorstore = Chroma(
                persist_directory=self.persist_dir,
                embedding_function=self.embeddings
            )
            print(f" База знаний загружена. Документов: {self.vectorstore._collection.count()}")
        else:
            print(" Создание новой базы знаний...")

            # Проверяем наличие файла с документами
            if not os.path.exists("car_knowledge.txt"):
                print(" Файл car_knowledge.txt не найден, создаю тестовый.")
                with open("car_knowledge.txt", "w", encoding="utf-8") as f:
                    f.write("""
Диагностика двигателя:
1. Если машина не заводится, проверьте аккумулятор (напряжение должно быть 12.5-12.7В)
2. Если стартер крутит, но двигатель не схватывает — проверьте топливную систему
3. Проблемы с иммобилайзером могут блокировать запуск

Замена масла:
1. Для бензиновых двигателей — каждые 8-10 тыс. км
2. Для дизельных — каждые 7-8 тыс. км
3. Используйте масло 5W30 или 5W40 в зависимости от сезона

Перегрев двигателя:
1. Проверьте уровень антифриза
2. Убедитесь, что работает вентилятор охлаждения
3. Возможна неисправность термостата

Тормозная система:
1. Тормозные колодки служат 20-30 тыс. км
2. При скрипе тормозов проверьте износ колодок
3. Тормозную жидкость менять каждые 2 года
""")

            # Загружаем и разбиваем документ
            loader = TextLoader("car_knowledge.txt", encoding='utf-8')
            docs = loader.load()

            splitter = RecursiveCharacterTextSplitter(
                chunk_size=300,
                chunk_overlap=50
            )
            chunks = splitter.split_documents(docs)

            # Создаём векторную базу
            self.vectorstore = Chroma.from_documents(
                documents=chunks,
                embedding=self.embeddings,
                persist_directory=self.persist_dir
            )
            print(f" База знаний создана. Фрагментов: {len(chunks)}")

        return self.vectorstore

    def _check_with_guardian(self, text_to_check, check_type="harm"):
        """
        Универсальная функция для проверки текста с помощью Granite Guardian.
        check_type может быть "harm", "jailbreak", "profanity" и т.д.
        Возвращает (is_unsafe, explanation)
        """
        # Формируем промпт для охранника
        prompt = f"""Проверь следующий текст на наличие {check_type}. Ответь только "Yes" или "No".

Текст: {text_to_check}

Результат:"""

        # Отправляем запрос к охраннику
        response = self.guard.invoke(prompt)
        result_text = response.strip().lower()

        # Определяем результат
        is_unsafe = "yes" in result_text[:5]  # проверяем первые 5 символов
        return is_unsafe, result_text

    def safe_ask(self, question):
        """
        Основной метод: поиск в RAG, генерация ответа и его проверка.
        """
        print(f"\n Обработка запроса: '{question}'")
        print("-" * 60)

        # 1. --- Input Guardrails: Проверяем сам вопрос на безопасность ---
        print(" Шаг 1: Проверка входа (Input Guardrail)...")
        is_query_unsafe, query_check_result = self._check_with_guardian(question, "harmful content")

        if is_query_unsafe:
            print(f"     Запрос содержит опасный контент: {query_check_result}")
            return {
                'question': question,
                'final_answer': " Запрос отклонён системой безопасности. Содержит потенциально опасный контент.",
                'blocked': True,
                'stage': 'input',
                'guardian_report': query_check_result,
                'latency': 0,
                'sources': [],
                'metrics': {
                    'hallucination_detected': False,
                    'hallucination_explanation': '',
                    'toxic_content': True
                }
            }
        print(f"     Запрос безопасен. ({query_check_result})")

        # 2. --- RAG: Ищем документы и генерируем ответ ---
        print(" Шаг 2: Поиск в базе знаний и генерация ответа (RAG)...")

        # Получаем векторную базу
        vectorstore = self._get_vectorstore()

        # Ищем релевантные документы
        docs = vectorstore.similarity_search(question, k=2)

        # Формируем контекст из документов
        context = "\n\n".join([doc.page_content for doc in docs])

        # Формируем промпт для основной LLM
        rag_prompt = f"""Используй информацию из документов ниже, чтобы ответить на вопрос.
Если ответа нет в документах, скажи "Я не знаю".

Документы:
{context}

Вопрос: {question}

Ответ:"""

        # Получаем ответ от основной модели
        start_time = datetime.now()
        llm_response = self.llm.invoke(rag_prompt)
        latency = (datetime.now() - start_time).total_seconds()

        print(f"     Ответ получен за {latency:.2f} сек.")
        print(f"     Сгенерированный ответ: {llm_response[:100]}...")

        # 3. --- Проверка на галлюцинации (мониторинг качества) ---
        print(" Шаг 3: Мониторинг качества (проверка на галлюцинации)...")

        # Простая проверка на галлюцинации: ответ не должен быть слишком длинным и должен содержать ключевые слова
        hallucination_detected = False
        hallucination_explanation = "OK"

        if len(llm_response) > 500:  # слишком длинный ответ - подозрительно
            hallucination_detected = True
            hallucination_explanation = "Ответ слишком длинный для данной базы знаний"
        elif "я не знаю" in llm_response.lower() and "документах" in llm_response.lower():
            # Если модель честно сказала, что не знает - это не галлюцинация
            hallucination_detected = False
        elif "аккумулятор" in question.lower() and "аккумулятор" not in llm_response.lower():
            # Если спрашивали про аккумулятор, а в ответе его нет
            hallucination_detected = True
            hallucination_explanation = "Ответ не соответствует контексту вопроса"

        print(f"    {' Галлюцинации обнаружены' if hallucination_detected else ' Галлюцинации не обнаружены'}")

        # 4. --- Output Guardrails: Проверяем ответ на токсичность ---
        print(" Шаг 4: Проверка выхода (Output Guardrail)...")
        is_unsafe_output, output_check_result = self._check_with_guardian(llm_response, "harmful content")

        if is_unsafe_output:
            print(f"     Ответ содержит опасный контент. Блокировка.")
            final_answer = " Ответ был заблокирован системой безопасности."
        else:
            print(f"     Ответ безопасен. ({output_check_result})")
            final_answer = llm_response

        # Формируем результат
        result = {
            'question': question,
            'final_answer': final_answer,
            'blocked': is_unsafe_output,
            'stage': 'output' if is_unsafe_output else 'none',
            'guardian_report': output_check_result if is_unsafe_output else 'OK',
            'latency': latency,
            'sources': [doc.metadata.get('source', 'unknown') for doc in docs],
            'metrics': {
                'hallucination_detected': hallucination_detected,
                'hallucination_explanation': hallucination_explanation,
                'toxic_content': is_unsafe_output,
            }
        }

        return result
```
