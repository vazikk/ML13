import ollama
from templates import *

def test_prompt(template_name, template, **kwargs):
    """Тестирует один промпт"""
    print(f"\n{'='*50}")
    print(f"ТЕСТИРУЕМ: {template_name}")
    print(f"{'='*50}")

    # Формируем промпт
    prompt = template.format(**kwargs)
    print(f"\n ПРОМПТ:\n{prompt}\n")

    # Отправляем в модель
    response = ollama.chat(model='llama3.1:8b', messages=[
        {'role': 'user', 'content': prompt}
    ])

    answer = response['message']['content']
    print(f" ОТВЕТ:\n{answer}\n")

    return answer

# Тест 1: Простой вопрос
test_prompt(
    "SIMPLE_QA",
    SIMPLE_QA,
    question="Что такое LLMOps?"
)

# Тест 2: С ролью эксперта
test_prompt(
    "ROLE_QA (эксперт)",
    ROLE_QA,
    role="опытный инженер по машинному обучению",
    question="Какие этапы включает LLMOps?"
)

# Тест 3: С ролью учителя (для сравнения)
test_prompt(
    "ROLE_QA (учитель)",
    ROLE_QA,
    role="учитель, который объясняет простыми словами",
    question="Какие этапы включает LLMOps?"
)

# Тест 4: Few-shot обучение
test_prompt(
    "FEW_SHOT",
    FEW_SHOT,
    examples="""
Вопрос: Что такое RAG?
Ответ: RAG (Retrieval-Augmented Generation) - это архитектура, которая combines поиск информации и генерацию текста.

Вопрос: Что такое fine-tuning?
Ответ: Fine-tuning - это процесс дообучения уже готовой модели на новых данных.
""",
    task="Объясни, что такое prompt engineering"
)

# Тест 5: С ограничениями
test_prompt(
    "CONSTRAINED",
    CONSTRAINED,
    question="Как работают большие языковые модели?",
    max_words=30,
    language="русский",
    extra_rule="Не используй технические термины"

from google.colab import files
from unsloth import FastLanguageModel
import torch
from datasets import Dataset
from trl import SFTTrainer
from transformers import TrainingArguments
import json

# 2. ЗАГРУЗКА ДАТАСЕТА
print(" Загрузите файл Dataset.json")
uploaded = files.upload()  # выбираем файл с датасетом

with open('Dataset.json', 'r') as f:
    data = json.load(f)
print(f" Загружено примеров: {len(data)}")

# 3. ФОРМАТИРОВАНИЕ ДАННЫХ ДЛЯ LLAMA 3.1 (ChatML формат)
def format_prompt(example):
    return f"""<|begin_of_text|><|start_header_id|>user<|end_header_id|>

{example['input']}<|eot_id|><|start_header_id|>assistant<|end_header_id|>

{example['output']}<|eot_id|>"""

formatted_data = [format_prompt(item) for item in data]
dataset = Dataset.from_dict({"text": formatted_data})

print(" Пример форматированного промпта:")
print(formatted_data[0][:200] + "...")

# 4. ЗАГРУЗКА БАЗОВОЙ МОДЕЛИ LLAMA 3.1
model_name = "unsloth/Llama-3.1-8B-bnb-4bit"
max_seq_length = 2048

model, tokenizer = FastLanguageModel.from_pretrained(
    model_name=model_name,
    max_seq_length=max_seq_length,
    load_in_4bit=True,  # 4-битная квантизация для экономии памяти
)

print(" Базовая модель загружена")

# 5. ДОБАВЛЕНИЕ LORA-АДАПТЕРОВ
model = FastLanguageModel.get_peft_model(
    model,
    r=16,  # ранг LoRA (16 оптимально)
    target_modules=[
        "q_proj", "k_proj", "v_proj", "o_proj",
        "gate_proj", "up_proj", "down_proj",
    ],
    lora_alpha=16,
    lora_dropout=0,
    bias="none",
    use_gradient_checkpointing="unsloth",
    random_state=3407,
)

print(f" LoRA адаптеры добавлены")
print(f"   Обучаемых параметров: {sum(p.numel() for p in model.parameters() if p.requires_grad):,}")

# 6. НАСТРОЙКА ОБУЧЕНИЯ
trainer = SFTTrainer(
    model=model,
    tokenizer=tokenizer,
    train_dataset=dataset,
    dataset_text_field="text",
    max_seq_length=max_seq_length,
    args=TrainingArguments(
        per_device_train_batch_size=2,
        gradient_accumulation_steps=4,
        num_train_epochs=3,  # 3 полных прохода по датасету
        learning_rate=2e-4,
        fp16=True,
        logging_steps=10,
        output_dir="outputs",
        report_to="none",
        save_strategy="epoch",
    ),
)

print(" НАЧАЛО ОБУЧЕНИЯ")
print("="*50)

# 7. ЗАПУСК ОБУЧЕНИЯ
trainer.train()

print("="*50)
print(" ОБУЧЕНИЕ ЗАВЕРШЕНО")

# 8. СОХРАНЕНИЕ МОДЕЛИ В ФОРМАТ GGUF ДЛЯ OLLAMA
model.save_pretrained_gguf(
    "llama3-car-final",
    tokenizer,
    quantization_method=["q4_k_m"]
)

print(" Модель сохранена в формате GGUF")
